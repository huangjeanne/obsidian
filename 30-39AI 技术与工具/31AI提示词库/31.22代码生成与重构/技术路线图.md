import graphviz

# --- Dimension 4: Visual Standards & Palette ---
COLOR_BLUE = "#0066CC"   # Main Process (Standard Task)
COLOR_GREEN = "#009966"  # Innovation Points (High Value)
COLOR_GREY = "#F0F0F0"   # Foundation/Plan B (Background)
COLOR_DECISION = "#FFFACD" # Decision Point
FONT_NAME = "SimHei"     # Sans-serif for Chinese (SimHei/Arial)

# Initialize Graphviz Digraph
dot = graphviz.Digraph('NSFC_Agri_Roadmap', comment='NSFC Technical Roadmap')

# --- Dimension 2: Structure & Aesthetics ---
# Top-to-Bottom layout, Orthogonal lines for professional look
dot.attr(rankdir='TB', splines='ortho', compound='true', nodesep='0.5', ranksep='0.6')
dot.attr('node', shape='box', style='filled', fontname=FONT_NAME, fontsize='12', margin='0.15,0.1')
dot.attr('edge', fontname=FONT_NAME, fontsize='10', color='#555555', arrowsize='0.8')

# Project Title
dot.node('Title', '项目名称：基于知识图谱与AI推理的数字农业决策支持系统研究', 
         shape='note', style='filled', fillcolor='#E0E0E0', fontsize='14', fontcolor='black', width='4')

# ==============================================================================
# PHASE I: 理论基础与数据构建 (RC1: Theory & Ontology)
# ==============================================================================
with dot.subgraph(name='cluster_0') as c:
    c.attr(label='第一阶段：多源异构知识解析与图谱构建 (Phase I: Theory & Data)', 
           style='dashed', color='#AAAAAA', fontcolor='#333333', bgcolor='#FAFAFA')

    # Input (Dimension 1: Start of Logical Loop)
    c.node('P1_Data', '输入: 多源异构农业数据\n(物联网传感、农学文献、遥感影像)', fillcolor=COLOR_GREY)

    # Process (Dimension 5: Granularity - Added specific params)
    c.node('P1_Ontology', '本体建模与语义一致性解析\n(参数: 实体对齐准确率>92%, 语义消歧F1>0.85)', 
           fillcolor=COLOR_BLUE, fontcolor='white')
    
    # Output of Phase I
    c.node('P1_KG', '构建农业领域知识图谱 (Agri-KG)\n(输出: 千万级实体规模, 三元组结构)', 
           fillcolor=COLOR_BLUE, fontcolor='white')

    # Edges inside Phase I
    c.edge('P1_Data', 'P1_Ontology', label='数据清洗/预处理')
    c.edge('P1_Ontology', 'P1_KG', label='图谱实例化')

# ==============================================================================
# PHASE II: 关键技术与模型研发 (RC2: Method & Algorithm)
# ==============================================================================
with dot.subgraph(name='cluster_1') as c:
    c.attr(label='第二阶段：图神经网络与推理模型研发 (Phase II: Core Innovation)', 
           style='dashed', color='#AAAAAA', fontcolor='#333333', bgcolor='#FAFAFA')

    # Baseline Method
    c.node('P2_GNN', '图神经网络(GNN)架构设计\n(技术: Heterogeneous Graph Attention)', 
           fillcolor=COLOR_BLUE, fontcolor='white')

    # --- Dimension 6: Innovation Highlight (Green + Star) ---
    c.node('P2_Innovation', '★ 融合小样本学习的作物关系推理模型\n(创新: Meta-learning策略, 样本量N<50)', 
           fillcolor=COLOR_GREEN, fontcolor='white')

    # --- Dimension 3: Time & Decision (Go/No-Go Diamond) ---
    c.node('P2_Decision', '模型性能评估\n(指标: 推理准确率>88%?)', 
           shape='diamond', style='filled', fillcolor=COLOR_DECISION, fontsize='10', height='0.8')

    # Plan B Loop
    c.node('P2_PlanB', 'Plan B: 引入专家规则约束\n(基于农学知识库的逻辑增强)', 
           fillcolor=COLOR_GREY, fontsize='10')

    # Output of Phase II
    c.node('P2_Output', '输出: 高鲁棒性智能推理引擎内核', fillcolor=COLOR_BLUE, fontcolor='white')

    # Edges inside Phase II
    c.edge('P2_GNN', 'P2_Innovation', label='特征融合')
    c.edge('P2_Innovation', 'P2_Decision', label='训练与测试')
    
    # Decision Logic
    c.edge('P2_Decision', 'P2_Output', label='是 (Yes)', color=COLOR_GREEN, style='bold')
    c.edge('P2_Decision', 'P2_PlanB', label='否 (No)', color='#CC0000', style='dashed')
    c.edge('P2_PlanB', 'P2_Innovation', label='反馈优化', style='dotted', constraint='false')

# ==============================================================================
# PHASE III: 应用验证与系统构建 (RC3: Application & Validation)
# ==============================================================================
with dot.subgraph(name='cluster_2') as c:
    c.attr(label='第三阶段：智能决策系统构建与验证 (Phase III: Validation)', 
           style='dashed', color='#AAAAAA', fontcolor='#333333', bgcolor='#FAFAFA')

    # System Integration (Dimension 5: Performance Params)
    c.node('P3_System', '智能决策支持系统集成\n(参数: 决策响应时间<500ms, 并发数>1000)', 
           fillcolor=COLOR_BLUE, fontcolor='white')

    # Domain Adaptation Scenarios
    c.node('P3_Verify', '典型数字农业场景验证\n(场景: 精准施肥推荐、病虫害预警)', 
           fillcolor=COLOR_BLUE, fontcolor='white')

    # Final Outcome
    c.node('P3_Final', '成果: 示范应用与评估报告\n(指标: 用户采纳率>80%, 成本降低15%)', 
           fillcolor=COLOR_GREEN, fontcolor='white')

    # Edges inside Phase III
    c.edge('P3_System', 'P3_Verify', label='系统部署')
    c.edge('P3_Verify', 'P3_Final', label='实地评估')

# ==============================================================================
# GLOBAL CONNECTIONS (Dimension 1 & 7: Logical Loop & Quality Control)
# ==============================================================================

# Connect Title to Start
dot.edge('Title', 'P1_Data', style='invis', minlen='1')

# Connect Phase I -> Phase II (KG feeds into GNN)
dot.edge('P1_KG', 'P2_GNN', label=' 知识图谱嵌入 ', 
         lhead='cluster_1', ltail='cluster_0', style='bold', color=COLOR_BLUE)

# Connect Phase II -> Phase III (Model feeds into System)
dot.edge('P2_Output', 'P3_System', label=' 算法集成 ', 
         lhead='cluster_2', ltail='cluster_1', style='bold', color=COLOR_BLUE)

# ==============================================================================
# LEGEND (Dimension 4: Visual Standards)
# ==============================================================================
with dot.subgraph(name='cluster_legend') as c:
    c.attr(label='图例 (Legend)', fontsize='10', style='solid', color='black', bgcolor='white')
    c.node('L_Main', '主线任务', fillcolor=COLOR_BLUE, fontcolor='white', width='1.5')
    c.node('L_Inno', '核心创新点 (★)', fillcolor=COLOR_GREEN, fontcolor='white', width='1.5')
    c.node('L_Base', '基础/备选方案', fillcolor=COLOR_GREY, width='1.5')
    c.node('L_Dec', '关键决策点', shape='diamond', fillcolor=COLOR_DECISION, width='1.5')
    
    # Invisible edges to stack legend items
    c.edge('L_Main', 'L_Inno', style='invis')
    c.edge('L_Inno', 'L_Base', style='invis')
    c.edge('L_Base', 'L_Dec', style='invis')

# Render the graph
try:
    # Saves source .gv and renders .png
    filename = 'NSFC_Agri_Roadmap'
    dot.render(filename, view=False, format='png', cleanup=True)
    print(f"✅ Success! Roadmap generated: {filename}.png")
except Exception as e:
    print(f"❌ Error: {e}")
    print("Ensure Graphviz is installed on your system (not just the python library).")
