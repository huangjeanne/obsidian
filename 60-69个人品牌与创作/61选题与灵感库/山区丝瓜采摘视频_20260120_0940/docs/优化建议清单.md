### 3. 实现硬字幕功能

**当前**: 字幕是独立SRT文件
**目标**: 将字幕烧录到视频中

**实施方案**:
```python
# 创建新文件: scripts/subtitle_renderer.py
from moviepy.editor import VideoFileClip, TextClip, CompositeVideoClip
import pysrt

def burn_subtitles(video_path, srt_path, output_path):
    """将SRT字幕硬编码到视频"""
    video = VideoFileClip(video_path)
    subs = pysrt.open(srt_path)

    # 生成字幕片段
    subtitles_clips = []
    for sub in subs:
        txt_clip = TextClip(
            sub.text,
            fontsize=24,
            color='white',
            font='PingFang-SC-Regular'
        ).set_position(('center', 'bottom')).set_start(
            sub.start.ordinal / 1000.0
        ).set_duration(
            (sub.end.ordinal - sub.start.ordinal) / 1000.0
        )
        subtitles_clips.append(txt_clip)

    # 合成
    final = CompositeVideoClip([video] + subtitles_clips)
    final.write_videofile(output_path)

# 命令行使用
if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--video", required=True)
    parser.add_argument("--srt", required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    burn_subtitles(args.video, args.srt, args.output)
```

**使用方法**:
```bash
python3 scripts/subtitle_renderer.py \
  --video 自动化采摘机器人_16镜头_完整动画.mp4 \
  --srt audio/subtitles.srt \
  --output 自动化采摘机器人_带字幕.mp4
```

### 4. 改进时长预测

**问题**: 目标60秒，实际37.92秒（-37%）
**原因**: 中文语速比预期快

**解决方案**:
```python
# 在script_generator.py中添加语速修正
def estimate_duration(text_chars, language='zh'):
    """
    估计文本朗读时长

    Args:
        text_chars: 字符数
        language: 语言（zh=中文, en=英文）

    Returns:
        估计秒数
    """
    if language == 'zh':
        # 中文平均语速: 5-6字/秒
        # 添加30%缓冲（停顿、语速变化）
        base_speed = 5.5
        buffer = 1.3
        return (text_chars / base_speed) * buffer
    else:
        # 英文平均语速: 3-4词/秒
        base_speed = 3.5
        buffer = 1.2
        return (text_chars / base_speed) * buffer
```

**所需时间**: 30分钟

```python
# 在video_workflow.py中添加
def auto_organize_project(topic, files_dict, base_path):
    """
    自动创建主题文件夹并整理文件

    Args:
        topic: 主题名称
        files_dict: 文件字典 {'video': path, 'script': path, ...}
        base_path: 基础路径（选题库）
    """
    import datetime
    from pathlib import Path
    import shutil

    # 创建文件夹
    timestamp = datetime.datetime.now().strftime("%Y%m%d")
    folder_name = f"{topic}-{timestamp}"
    project_path = Path(base_path) / folder_name
    project_path.mkdir(parents=True, exist_ok=True)

    # 创建子目录
    (project_path / "audio").mkdir(exist_ok=True)
    (project_path / "visuals").mkdir(exist_ok=True)
    (project_path / "docs").mkdir(exist_ok=True)

    # 复制文件
    for category, file_path in files_dict.items():
        if category == 'audio':
            shutil.copy(file_path, project_path / "audio")
        elif category == 'visuals':
            shutil.copy(file_path, project_path / "visuals")
        # ...其他分类

    # 更新项目索引
    index_file = Path(base_path) / "视频项目索引.md"
    # ...更新索引

    return project_path
```

```python
# 修改animated_video_composer.py
# 在生成镜头时，为每个镜头调用cover_generator生成独特的图像

from cover_generator import GeminiImageGenerator

def generate_scene_images(scenes_data, output_dir):
    """
    为每个场景生成独特的图像

    Args:
        scenes_data: 场景数据列表
        output_dir: 输出目录
    """
    generator = GeminiImageGenerator(model="nano-banana-pro-preview")

    for i, scene in enumerate(scenes_data, 1):
        # 提取视觉描述
        visual_desc = scene.get('visual_description', '')

        # 生成图像
        output_path = Path(output_dir) / f"scene_{i:03d}.png"
        result = generator.generate_image(
            prompt=visual_desc,
            output_path=output_path
        )

        if result.success:
            print(f"✅ 场景{i}图像生成成功")
        else:
            print(f"❌ 场景{i}图像生成失败: {result.error_message}")
```

**所需时间**: 2-3小时（取决于场景数量）
